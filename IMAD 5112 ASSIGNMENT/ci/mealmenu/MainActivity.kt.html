package vcmsa.ci.mealmenu

import android.annotation.SuppressLint
import android.os.Bundle
import android.view.View
import android.widget.Button
import android.widget.EditText
import android.widget.TextView
import androidx.activity.enableEdgeToEdge
import androidx.appcompat.app.AppCompatActivity
import androidx.core.view.ViewCompat
import androidx.core.view.WindowInsetsCompat
import java.util.Locale

class MainActivity : AppCompatActivity() {

    var inputTime : EditText? = null
    var SuggestedMeal : Button? = null
    var output : TextView? = null
    var ResetButton : Button? = null
//(OpenAI.2025.Meta AI.Llama 3 version.Natural language processing and machine learning algorithms.Web address of Meta AI unavailable.)

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        enableEdgeToEdge()
        setContentView(R.layout.activity_main)

        inputTime = findViewById<View>(R.id.inputTime) as EditText
        SuggestedMeal = findViewById<View>(R.id.SuggestedMeal) as Button
        output = findViewById<View>(R.id.output) as TextView
        ResetButton = findViewById<View>(R.id.ResetButton) as Button
//(OpenAI.2025.Meta AI.Llama 3 version.Natural language processing and machine learning algorithms.Web address of Meta AI unavailable.)

        SuggestedMeal!!.setOnClickListener {
            val time =
                inputTime!!.text.toString().lowercase(Locale.getDefault())
            var meal =  when (time) {
                "morning" -> "meal : Eggs"
                "mid morning" -> "meal : Fruit"
                "afternoon" ->  "meal : Sandwich"
                "mid-afternoon" -> "meal : Cake"
                "dinner" -> "meal : Pasta"
                "afternoon" -> "meal : IceCream"

                else -> "meal:Apologies but please find a menu within the suggestions above!"

            }//(OpenAI.2025.Meta AI.Llama 3 version.Natural language processing and machine learning algorithms.Web address of Meta AI unavailable.)

            output!!.text = meal
        }//(OpenAI.2025.Meta AI.Llama 3 version.Natural language processing and machine learning algorithms.Web address of Meta AI unavailable.)

        ResetButton!!.setOnClickListener{
            inputTime!!.text.clear()
            output!!.text= ""
        }//(OpenAI.2025.Meta AI.Llama 3 version.Natural language processing and machine learning algorithms.Web address of Meta AI unavailable.)

        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main)) { v, insets ->
            val systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars())
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom)
            insets
        }
    }
}
